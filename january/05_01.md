Solution

```js
function bitwiseAND(n1, n2) {
  return n1&n2;
}

function bitwiseOR(n1, n2) {
	  return n1|n2;
}

function bitwiseXOR(n1, n2) {
	  return n1^n2;
}
```

Solution

```js
function minMax(arr) {
  const min = Math.min(...arr);
  const max = Math.max(...arr);
  
  return [min, max];
}
```
